#include(CPU.txt)
#include(system.txt)
#include(system.ASCII.txt)

:one
0				// i loop start
10				// i loop end 
DO				// for i
    >r				// save i
    system.ascii.a r@ +		// j loop start
    system.ascii.z 1 + 9 - r@ +		// j loop end
    DO	// for j
	dup system.emit
    LOOP
    system.ascii.cr 0x1000 !io 
    system.ascii.nl 0x1000 !io 
    r>				// restore loop var, clear return stack
LOOP
:end BRA:end

BRA:one


:start
0x40	// @
:nextChar
dup 0x1000 !io 

dup 0x60 = IF 
     drop 
     BRA:start 
ENDIF
1 +
BRA:nextChar
BRA:start
